def ensure_value_at_path(klass, path, value)
  path.each_with_index.inject(self) { |obj, (attr, index)|
    if index == (path.size - 1)
      obj.send("#{attr}=", value)
    else
      if attr.class == Fixnum
        if obj.send(:at, attr).nil?
          obj.append(Object::const_get("Kubeclient::#{klass}").new)
        end
        obj.send(:at, attr)
      else
        obj.send(attr)
      end
    end
  }
end
